64  const std::vector<double>& x,
65  const std::vector<double>& y,
66  std::string method,
67  std::vector<double> weights = std::vector<double>())
68 {
69  // determine effective sample size
70  double n_eff;
71  if (weights.size() == 0) {
72  n_eff = static_cast<double>(x.size());
73  } else {
74  n_eff = wdm_utils::effective_sample_size(weights);
75  }
76 
77  // calculate test statistic
78  double stat;
79  if (method == "hoeffd") {
80  stat = hoeffd(x, y, weights) / 30.0 + 1.0 / (36.0 * n_eff);
81  } else if (method == "ktau") {
82  stat = ktau(x, y, weights);
83  stat *= std::sqrt(9 * n_eff / 4);
84  } else if (method == "prho") {
85  stat = boost::math::atanh(prho(x, y, weights));
86  stat *= std::sqrt(n_eff - 3);
87  } else if (method == "srho") {
88  stat = boost::math::atanh(srho(x, y, weights));
89  stat *= std::sqrt((n_eff - 3) / 1.06);
90  } else if (method == "bbeta") {
91  stat = bbeta(x, y, weights);
92  stat *= std::sqrt(n_eff);
93  } else {
94  throw std::runtime_error("method not impolemented.");
95  }
96 
97  return stat;
98 }
